use POSTMAN; put all (key,value) pairs in the 'Body' tab


http://127.0.0.1:8000/account/	[AccountView]

Login [GET]
	'login', username, password

Logout [GET]
	'logout'

Create Account [POST]
	'create', username, password, email, name, phone_number, category

Delete Account [POST]
	'delete', username


http://127.0.0.1:8000/group/	[GroupView]

Group Name [GET]
	'groupname', groupid

Group ID [GET]
	'groupid', groupname

Find user's groups [GET]
	'grouplist', userid

List Group Members [GET]
	'groupmembers', groupid 

Create a new group [POST]
	'create', userid, groupname

Delete a group [POST]
	'delete', groupid

Add Member to Group [POST]
	'addmember', groupid, userid

Delete Member from Group [POST]
	'deletemember', groupid, userid


http://127.0.0.1:8000/user/		[UserView]

User Name [GET]
	'username', userid

User ID [GET]
	'userid', username


http://127.0.0.1:8000/friend/	[FriendView]

Friends List [GET]
	'friendlist', userid
